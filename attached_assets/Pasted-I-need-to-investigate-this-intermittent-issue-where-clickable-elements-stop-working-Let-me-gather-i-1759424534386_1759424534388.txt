The Solution
Created centralized logout helper (client/src/lib/auth-helpers.ts):

export async function handleLogout(setUserId?: (id: string | null) => void) {
  await supabase.auth.signOut();  // Supabase handles its own cleanup
  if (setUserId) setUserId(null);
  window.location.href = '/';     // Clean redirect
}
