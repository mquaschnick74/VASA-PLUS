**Step 1: Create new file `server/services/literary-patterns.ts`**

```typescript
// Literary Psychological Patterns from Classic Literature
// Separate module to keep main pattern service manageable

import { PatternMatch } from './css-pattern-service';

export const LITERARY_PATTERNS = {
  EXISTENTIAL_CRISIS: [
    /what('?s| is) the point (of|in)/gi,
    /nothing matters anymore/gi,
    /life has no (meaning|purpose)/gi,
    // ... rest of patterns
  ],
  
  MORAL_TORMENT: [
    /I deserve (this|punishment|suffering)/gi,
    /can('?t| not) forgive myself/gi,
    // ... rest of patterns
  ],
  
  EPISTEMIC_DOUBT: [
    /the more I (learn|know).*the less I/gi,
    // ... rest of patterns
  ],
  
  KAFKA_ALIENATION: [
    /no one understands what I'?m going through/gi,
    // ... rest of patterns
  ],
  
  SOCIAL_MASKING: [
    /I (have to|must) pretend to be/gi,
    // ... rest of patterns
  ]
};

// Export detection function
export function detectLiteraryPatterns(
  transcript: string,
  detectPatternCategory: Function,
  debug: boolean = false
) {
  return {
    existentialPatterns: detectPatternCategory(transcript, LITERARY_PATTERNS.EXISTENTIAL_CRISIS, debug),
    moralTormentPatterns: detectPatternCategory(transcript, LITERARY_PATTERNS.MORAL_TORMENT, debug),
    epistemicPatterns: detectPatternCategory(transcript, LITERARY_PATTERNS.EPISTEMIC_DOUBT, debug),
    kafkaPatterns: detectPatternCategory(transcript, LITERARY_PATTERNS.KAFKA_ALIENATION, debug),
    socialMaskingPatterns: detectPatternCategory(transcript, LITERARY_PATTERNS.SOCIAL_MASKING, debug)
  };
}
```
