// Add narrative-specific pattern detection
const NARRATIVE_PATTERNS = {
  STORY_CONTRADICTION: [
    /one part of me (?:says|thinks|believes) (.+) but another (?:says|thinks|believes) (.+)/i,
    /my story (?:says|tells) (.+) but (?:I|my) (.+)/i,
    /I tell myself (.+) but (?:then I|I also) (.+)/i
  ],
  NARRATIVE_GAP: [
    /the story I tell (?:myself|others) (?:is|isn't) (.+)/i,
    /(?:I|my) narrative (?:doesn't|does not) match (.+)/i,
    /there's a gap between what I (?:say|tell) and (.+)/i
  ],
  FRAGMENTED_STORY: [
    /(?:I|my story) (?:doesn't|does not) make sense/i,
    /(?:I|it) (?:feels|seems) (?:scattered|fragmented|broken)/i,
    /can't (?:find|create|tell) (?:a|my) (?:coherent|clear) story/i
  ]
};

// Update the main detection function
export async function detectEnhancedCSSPatterns(
  transcript: string,
  userId?: string
): Promise<any> {
  const existingPatterns = await detectCSSPatterns(transcript);
  
  // Calculate narrative fragmentation score (0-10)
  let narrativeFragmentation = 0;
  
  // Check for story contradictions
  for (const pattern of NARRATIVE_PATTERNS.STORY_CONTRADICTION) {
    if (pattern.test(transcript)) {
      narrativeFragmentation += 3;
    }
  }
  
  // Check for narrative gaps
  for (const pattern of NARRATIVE_PATTERNS.NARRATIVE_GAP) {
    if (pattern.test(transcript)) {
      narrativeFragmentation += 2;
    }
  }
  
  // Check for fragmented stories
  for (const pattern of NARRATIVE_PATTERNS.FRAGMENTED_STORY) {
    if (pattern.test(transcript)) {
      narrativeFragmentation += 4;
    }
  }
  
  // Cap at 10
  narrativeFragmentation = Math.min(narrativeFragmentation, 10);
  
  // Add narrative significance to patterns
  const enhancedPatterns = existingPatterns.map(pattern => ({
    ...pattern,
    narrativeSignificance: {
      storyFragment: pattern.content,
      symbolicDensity: calculateSymbolicDensity(pattern.content),
      temporalOrientation: determineTemporalOrientation(pattern.content),
      fragmentationScore: narrativeFragmentation
    }
  }));
  
  // Store high-significance narrative markers
  if (userId && narrativeFragmentation > 6) {
    await supabase.from('therapeutic_context').insert({
      user_id: userId,
      context_type: 'narrative_marker',
      content: `High narrative fragmentation detected: ${narrativeFragmentation}/10`,
      importance: narrativeFragmentation,
      pattern_type: 'NARRATIVE_FRAGMENTATION'
    });
  }
  
  return {
    patterns: enhancedPatterns,
    narrativeFragmentation,
    suggestedAgent: narrativeFragmentation > 7 ? 'zhanna' : null
  };
}

// Helper functions
function calculateSymbolicDensity(content: string): number {
  // Words that indicate high symbolic density
  const symbolicMarkers = [
    'always', 'never', 'everything', 'nothing', 'everyone',
    'can\'t', 'must', 'should', 'have to', 'need to',
    'die', 'kill', 'destroy', 'save', 'rescue'
  ];
  
  let density = 0;
  const words = content.toLowerCase().split(/\s+/);
  
  for (const marker of symbolicMarkers) {
    if (words.includes(marker)) density += 2;
  }
  
  return Math.min(density, 10);
}

function determineTemporalOrientation(content: string): 'past' | 'present' | 'future' {
  const pastMarkers = /(?:was|were|had|used to|before|ago|last|previous)/i;
  const futureMarkers = /(?:will|going to|gonna|soon|tomorrow|next|later)/i;
  
  if (pastMarkers.test(content)) return 'past';
  if (futureMarkers.test(content)) return 'future';
  return 'present';
}

import { determineNarrativePhase } from '../../client/src/lib/config/vapi-agents-enhanced';

export async function analyzeAndSuggestAgent(
  transcript: string,
  currentAgent: string,
  userId?: string
): Promise<any> {
  // Get enhanced patterns with narrative fragmentation
  const { patterns, narrativeFragmentation } = await detectEnhancedCSSPatterns(transcript, userId);
  
  // Count pattern types
  const patternCounts = {
    cvdc: patterns.filter(p => p.type === 'CVDC').length,
    ibm: patterns.filter(p => p.type.includes('IBM')).length,
    thend: patterns.filter(p => p.type === 'Thend').length,
    crisis: patterns.some(p => p.crisis_flag),
    distressLevel: Math.max(...patterns.map(p => p.emotional_intensity || 0)),
    narrativeFragmentation // Add the new score
  };
  
  // Get user's narrative phase if we have session history
  let narrativePhase = 'building';
  if (userId) {
    const sessions = await supabase
      .from('therapeutic_sessions')
      .select('*')
      .eq('user_id', userId)
      .order('created_at', { ascending: false })
      .limit(10);
    
    const sessionCount = sessions.data?.length || 0;
    const allPatterns = sessions.data?.flatMap(s => s.metadata?.patterns || []) || [];
    narrativePhase = determineNarrativePhase(sessionCount, allPatterns);
  }
  
  // Select agent based on enhanced criteria
  const suggestedAgent = selectAgentForPattern(patternCounts);
  const shouldSwitch = shouldSwitchAgent(currentAgent, patternCounts);
  
  // Store the analysis
  if (userId) {
    await supabase.from('therapeutic_sessions').update({
      metadata: {
        ...existingMetadata,
        narrative_phase: narrativePhase,
        narrative_fragmentation: narrativeFragmentation,
        last_analysis: new Date().toISOString()
      }
    }).eq('user_id', userId).eq('status', 'active');
  }
  
  return {
    suggestedAgent,
    shouldSwitch,
    patterns,
    narrativeFragmentation,
    narrativePhase,
    reason: determineReason(patternCounts, narrativePhase)
  };
}

function determineReason(patterns: any, phase: string): string {
  if (patterns.narrativeFragmentation > 7) {
    return 'High narrative fragmentation requires grounding';
  }
  if (patterns.crisis) {
    return 'Crisis indicators detected';
  }
  if (patterns.thend > 2) {
    return 'Integration moment emerging';
  }
  if (phase === 'integrating') {
    return 'Supporting narrative integration phase';
  }
  return 'Continuing narrative exploration';
}
```

export async function buildEnhancedMemoryContext(userId: string): Promise<string> {
  const userData = await getUserData(userId);
  
  // Get narrative markers
  const narrativeMarkers = await supabase
    .from('therapeutic_context')
    .select('*')
    .eq('user_id', userId)
    .in('context_type', ['narrative_marker', 'narrative_theme'])
    .order('importance', { ascending: false })
    .limit(5);
  
  // Get session progression
  const sessions = await supabase
    .from('therapeutic_sessions')
    .select('metadata')
    .eq('user_id', userId)
    .order('created_at', { ascending: false })
    .limit(3);
  
  // Build narrative-aware context
  let context = '';
  
  // Frame as continuing story
  if (userData.sessionCount > 0) {
    context += `Continuing ${userData.firstName}'s therapeutic narrative. `;
    context += `Session ${userData.sessionCount + 1} of their journey. `;
  }
  
  // Add narrative themes if present
  if (narrativeMarkers.data?.length > 0) {
    context += '\n\nKey narrative themes from previous work: ';
    narrativeMarkers.data.forEach(marker => {
      if (marker.pattern_type === 'NARRATIVE_FRAGMENTATION') {
        context += `Story showing fragmentation. `;
      } else {
        context += `${marker.content}. `;
      }
    });
  }
  
  // Add phase information
  const lastSession = sessions.data?.[0];
  if (lastSession?.metadata?.narrative_phase) {
    context += `\n\nNarrative journey is in ${lastSession.metadata.narrative_phase} phase.`;
  }
  
  // Add pattern evolution
  if (userData.dominantPatterns?.length > 0) {
    context += '\n\nStories being explored: ';
    userData.dominantPatterns.forEach(pattern => {
      if (pattern === 'CVDC') context += 'contradictory narratives, ';
      if (pattern === 'IBM') context += 'story-behavior gaps, ';
      if (pattern === 'Thend') context += 'emerging integration, ';
    });
  }
  
  return context;
}
```
