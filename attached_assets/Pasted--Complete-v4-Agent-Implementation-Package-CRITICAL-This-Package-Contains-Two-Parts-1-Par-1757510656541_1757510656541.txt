export function selectAgentForPatterns(patterns: {
  cvdc: number;
  ibm: number;
  thend: number;
  crisis: boolean;
  distressLevel: number;
}): string {
  // Crisis/high distress ALWAYS triggers Zhanna
  if (patterns.crisis || patterns.distressLevel >= 7) {
    return 'zhanna';
  }
  
  // Integration moments
  if (patterns.thend > 2) {
    return 'marcus';
  }
  
  // Behavioral patterns
  if (patterns.ibm > patterns.cvdc && patterns.ibm > 3) {
    return 'mathew';
  }
  
  // Contradictions
  if (patterns.cvdc > 2) {
    return 'sarah';
  }
  
  // Default
  return 'sarah';
}

async function analyzeForAgentSuggestionEnhanced(
  patterns: any[],

): Promise<string | null> {
  
  // CRITICAL: Crisis detection FIRST
  const crisisPattern = patterns.find(p => 
    p.metadata?.crisis_flag || 
    p.metadata?.distress_level >= 7 ||
    p.stage === 'CRISIS'
  );
  
  if (crisisPattern && currentAgent !== 'zhanna') {
    console.log('🚨 Crisis detected - suggesting Zhanna');
    return 'zhanna';
  }
  
  // If currently with Zhanna, check if stable for handoff
  if (currentAgent === 'zhanna') {
    const latestDistress = patterns[patterns.length - 1]?.metadata?.distress_level || 10;
    if (latestDistress >= 5) {
      return null; // Stay with Zhanna
    }
    // Ready for handoff - determine next agent
    const cvdcCount = patterns.filter(p => p.stage === 'CVDC').length;
    const ibmCount = patterns.filter(p => p.metadata?.pattern_type === 'IBM').length;
    const thendCount = patterns.filter(p => p.stage === 'THEND').length;
    
    if (thendCount > 0) return 'marcus';
    if (ibmCount > cvdcCount) return 'mathew';
    if (cvdcCount > 0) return 'sarah';
    return null; // Let user choose
  }

// ADD crisis detection to detectEnhancedCSSPatterns:

export async function detectEnhancedCSSPatterns(
  content: string,
  speakerType: 'user' | 'assistant'
): Promise<CSSPattern[]> {
  const patterns: CSSPattern[] = [];
  
  // CRISIS DETECTION - HIGHEST PRIORITY
  const crisisIndicators = [
    'panic', 'can\'t breathe', 'chest tight', 'heart racing',
    'dizzy', 'unreal', 'floating', 'disconnected',
    'hurt myself', 'want to die', 'not safe', 'losing control'
  ];
  
  const contentLower = content.toLowerCase();
  const hasCrisis = crisisIndicators.some(indicator => 
    contentLower.includes(indicator)
  );
  
  // Check for distress level mentions
  const distressMatch = content.match(/(\d+)\s*(?:\/|out of)\s*10/);
  const distressLevel = distressMatch ? parseInt(distressMatch[1]) : null;
  
  if (hasCrisis || (distressLevel && distressLevel >= 7)) {
    patterns.push({
      stage: 'CRISIS',
      confidence: 1.0,
      metadata: {
        crisis_flag: true,
        distress_level: distressLevel || 10,
        suggested_agent: 'zhanna',
        priority: 'immediate'
      }
    });
    return patterns; // Return immediately for crisis
  }
  
  // ... rest of existing pattern detection
}
